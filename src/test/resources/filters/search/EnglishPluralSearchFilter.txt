String[] words = name.split(" ");
StringBuffer buff = new StringBuffer();
for(int i=0; i < words.length; i++){
    if(words[i].endsWith("y"))
        buff.append(words[i]+"ies");
    else
        buff.append(words[i]+"s");
}
return !nameDao.find(buff.toString()).isEmpty();
